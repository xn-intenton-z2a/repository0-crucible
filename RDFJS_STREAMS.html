Supplementary Details

Default parameters: StreamPipelineOptions.maxConcurrency=Infinity; delayErrors=false. Backpressure: next callback signals readiness for next quad. Error handling: pipeline rejects on first error unless delayErrors=true. Node.js stream compatibility: wrap Source with Readable.from for integration. Import patterns: sink.import returns completion Promise. DataFactory implementations must preserve termType and value properties. Dataset.match supports undefined args as wildcards.

function arrayToStream(quads:Q[]):Source { let index=0; return function push(quad, next){ if(index<quads.length){ next(); quad=quads[index++]; this(quad,next);} else { this(null);} } }

async function streamToArray(source:Source):Promise<Q[]> { const result:Q[]=[]; return new Promise((resolve,reject)=>{ function sink(quad:Q|null){ if(quad){ result.push(quad);} else { resolve(result);} } source(null); source=source; }); }

async function pipeline(source:Source, ...transforms:Array<(quad:Q)=>Q>, sink:Sink):Promise { let transformedSource:Source=source; for(const fn of transforms){ transformedSource=(quad, next)=>{ const out=quad!==null?fn(quad):null; return transformedSource(out,next); }; } return sink.import(transformedSource); }
